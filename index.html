Code Example

@Injectable({
  providedIn: "root"
})
export class AuthService {

  public isAuth: boolean = false;

  public isAuth$: Observable;
  public isAuthSubject$: Subject;

  constructor(
    private userService: UserService,
    private apiService: ApiService,
  ) {
     this.isAuthSubject$ = new Subject();
     this.isAuth$ = this.isAuthSubject$.asObservable();
     this.isAuthSubject$.next(this.isAuth);
  }

  isAuthenticated(): Observable {
    return this.isAuth$;
  }

  login(email, password): Observable {
    return from(
      this.apiService.getUserTokenByCredentials(email, password).pipe(
       tap((data: any) => {
       this.isAuth = true;
       this.isAuthSubject$.next(this.isAuth);
       return data;
     }),
     catchError( err => {
         return throwError(err);
     })));
  }

  logout(): Observable {
      this.isAuth = false;
      this.isAuthSubject$.next(this.isAuth);
      return this.isAuth$;
  }
}
